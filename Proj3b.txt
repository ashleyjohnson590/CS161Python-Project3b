#Description: asks user for positive integer, then then prints a list of all
# positive integers that divide that number evenly,including itself and 1, in
# ascending order.

number = int(input("Please enter a positive integer: "))
nums=1

print("The factors of ", number, "are:")
while(nums<=number):
    if number % nums == int():
        print(nums)
    nums+=1
